Pi.createPayment({ amount: 1.25, memo: "شراء من تطبيق أرابيكا", metadata: { auth: "arabeka" }, }, { // 1️⃣ الخطوة الأولى: إرسال الطلب لملف approve.js للموافقة onReadyForServerApproval: function(paymentId) { fetch('/api/approve', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ paymentId: paymentId }) }) .then(response => response.json()) .then(data => console.log("Approved on server:", data)) .catch(err => console.error("Approval error:", err)); }, // 2️⃣ الخطوة الثانية: إرسال الـ txid لملف complete.js بعد خصم المحفظة لتأكيد المعاملة onReadyForServerCompletion: function(paymentId, txid) { fetch('/api/complete', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ paymentId: paymentId, txid: txid }) }) .then(response => response.json()) .then(data => { console.log("Completed on server:", data); alert("تمت عملية الدفع بنجاح كامل!"); }) .catch(err => console.error("Completion error:", err)); }, // 3️⃣ الخطوة الثالثة: إذا ألغى المستخدم الدفع onCancel: function(paymentId) { console.log("Payment canceled by user:", paymentId); }, // 4️⃣ الخطوة الرابعة: في حال حدوث أي خطأ مفاجئ onError: function(error, paymentId) { console.error("Payment error occurred:", error); }, // 5️⃣ الخطوة الخامسة (الحل السحري): التعامل التلقائي مع المدفوعات المعلقة وإغلاقها onIncompletePaymentFound: function(payment) { console.log("Found an incomplete payment, resolving now...", payment.id); fetch('/api/complete', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ paymentId: payment.id, txid: payment.txid }) }) .then(response => response.json()) .then(data => { console.log("Incomplete payment resolved:", data); alert("تمت تسوية المعاملة المعلقة السابقة بنجاح! يمكنك إعادة المحاولة الآن."); }) .catch(err => console.error("Error resolving incomplete payment:", err)); } });
